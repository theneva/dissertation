    var sumFrom1ToLimit = function(limit, plusFunction) {
    var sum = 0;

    for (var i = 1; i <= limit; i++) {
        sum = plusFunction(sum, i); // !
    }

    return sum;
}

// 1 + 2 + 3 + ... + 10
